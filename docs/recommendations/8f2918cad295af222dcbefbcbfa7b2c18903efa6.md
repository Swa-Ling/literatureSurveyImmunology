                });

                // keep the width of the select element same as the column
                select.style.width = '100%';
 
                // Add list of options

                    .sort()

                        select.add(new Option(d));
                    });
            });
    },

      // Open row.

      row.child(content).show();
      tr.addClass('shown');
      tr.find('td:first-child').html('<i class="material-icons">visibility</i>');
    }
  });
